        % Button pushed function: ChooseFileButton
        function ChooseFileButtonPushed(app, event)
        global data;
        [filename, filepath] = uigetfile('*.csv', 'Select CSV file');
        
        % Check if a file was selected
        if isequal(filename, 0)
            disp('No file selected.');
            return;
        end
        
        % Read data from the CSV file
        data = readmatrix(fullfile(filepath, filename));
        end

        % Button pushed function: PlotthedataButton
        function PlotthedataButtonPushed(app, event)
           global data;
           global fs;
           global t;
           fs = 1000;  % Sample rate 
            t = (0:length(data)-1) / fs;  % Time vector
            plot(app.UIAxes,t,data)
            xlabel(app.UIAxes,'Time (s)');
            ylabel(app.UIAxes,'Amplitude');
            title(app.UIAxes,'ECG Signal - Time Domain');
            Y = fft(data);
            N = length(data);
            f = (0:N-1) * (fs / N);  % Frequency vector
            P = abs(Y).^2 / N;  % Power spectrum
            plot(app.UIAxes2,f, P);
            xlabel(app.UIAxes2,'Frequency (Hz)');
            ylabel(app.UIAxes2,'Power');
            title(app.UIAxes2,'ECG Signal - Frequency Domain');
            xlim(app.UIAxes2,[0 100])
        end

        % Button pushed function: 
        % ComputethecharacteristicsofthedataButton
        function ComputethecharacteristicsofthedataButtonPushed(app, event)
            global fs;
            global data;
            global t;
            order = 4;  % Filter order
            cutoff_freq = 50;  % Cutoff frequency in Hz
            [b_butter, a_butter] = butter(order, cutoff_freq / (fs/2));
            filtered_butter = filtfilt(b_butter, a_butter, data);
            mean_filtered = mean(filtered_butter);
            median_filtered = median(filtered_butter);
            variance_filtered = var(filtered_butter);
            window = hamming(length(data));
            [Pxx_original, f_original] = periodogram(data, window, [], fs, 'onesided');
            [Pxx_filtered, f_filtered] = periodogram(filtered_butter, window, [], fs, 'onesided');
            plot(app.UIAxes2_2,t, filtered_butter);
            title(app.UIAxes2_2,'Filtered (Butterworth)');
            ylabel(app.UIAxes2_2,'Amplitude');
            xlabel(app.UIAxes2_2,'Time (s)');
            plot(app.UIAxes_2,f_original, 10*log10(Pxx_original));
            title(app.UIAxes_2,'Spectral Characteristics (Original)');
            ylabel(app.UIAxes_2,'Power (dB)');
            xlabel(app.UIAxes_2,'Frequency (Hz)');
            app.ValueLabel.Text=sprintf('%.2f',  mean_filtered)
            app.ValueLabel_2.Text=sprintf('%.2f',  median_filtered)
            app.ValueLabel_3.Text=sprintf('%.2f',  variance_filtered)
        end
    end
